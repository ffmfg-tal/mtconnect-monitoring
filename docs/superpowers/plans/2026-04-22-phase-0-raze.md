- [ ] **Step 2: If broken imports exist, rewrite the test to stand alone**

If the test used `setupD1` / `seedMachines` helpers to exercise auth via ingest routes, replace with a standalone test that calls `requireEdgeSecret` directly. Template:

```typescript
import { describe, it, expect } from "vitest";
import { Hono } from "hono";
import { requireEdgeSecret } from "../src/auth";
import type { Env } from "../src/types";

function makeApp(secret: string) {
  const app = new Hono<{ Bindings: Env }>();
  app.use("*", requireEdgeSecret);
  app.get("/protected", (c) => c.json({ ok: true }));
  return app;
}

describe("requireEdgeSecret", () => {
  const env = { EDGE_SHARED_SECRET: "correct-secret" } as Env;

  it("401s when no X-Edge-Secret header", async () => {
    const res = await makeApp(env.EDGE_SHARED_SECRET).fetch(
      new Request("http://test/protected"),
      env,
    );
    expect(res.status).toBe(401);
  });

  it("401s on wrong secret", async () => {
    const res = await makeApp(env.EDGE_SHARED_SECRET).fetch(
      new Request("http://test/protected", {
        headers: { "X-Edge-Secret": "wrong" },
      }),
      env,
    );
    expect(res.status).toBe(401);
  });

  it("passes with correct secret", async () => {
    const res = await makeApp(env.EDGE_SHARED_SECRET).fetch(
      new Request("http://test/protected", {
        headers: { "X-Edge-Secret": "correct-secret" },
      }),
      env,
    );
    expect(res.status).toBe(200);
  });

  it("500s when server misconfigured", async () => {
    const res = await makeApp("").fetch(
      new Request("http://test/protected", {
        headers: { "X-Edge-Secret": "anything" },
      }),
      { EDGE_SHARED_SECRET: "" } as Env,
    );
    expect(res.status).toBe(500);
  });
});
```
